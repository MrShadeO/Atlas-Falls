get_lower_house_required_seats_10_percent = {
    custom_trigger_tooltip = {
        tooltip = custom_trg_check_lower_party_seats_10
        OR = {
            AND = {
                has_government = anarchism_ideology

                add_to_temp_variable = { temp_anarchism_seats = coalition_partner_lower_house_seats }

                    var = temp_anarchism_seats
