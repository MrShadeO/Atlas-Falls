defined_text = {
	name = GetInvestmentPPCost
	text = {
		trigger = {
			set_temp_variable = {
				var = temp_corp_investment_actual_cost
				value = corp_investment_cost_var
			}
			add_to_temp_variable = { temp_corp_investment_actual_cost = 1 }

			has_political_power > corp_investment_cost_var
		}
		localization_key = STANDARD_INVESTMENT_COST_CAN_AFFORD
	}
	text = {
		trigger = {
			NOT = { has_political_power > corp_investment_cost_var }
		}
		localization_key = STANDARD_INVESTMENT_COST_CANNOT_AFFORD
	}
	text = {
		localization_key = ERROR
	}
}
