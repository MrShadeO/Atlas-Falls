set_government_federal_republic = {
    custom_effect_tooltip = CHANGE_TO_FEDERAL_REPUBLIC
    hidden_effect = {
        clr_country_flag = is_parliamentary_republic
        clr_country_flag = is_oligarchy
        clr_country_flag = is_socialist_republic
        clr_country_flag = is_constitutional_monarchy
        clr_country_flag = is_absolute_monarchy
        clr_country_flag = is_dictatorship
        clr_country_flag = is_one_party_state
        clr_country_flag = is_anarcho_communist
        clr_country_flag = is_anarcho_capitalist
        clr_country_flag = is_provisional_government
        clr_country_flag = is_colonial_government

        set_country_flag = is_federal_republic

        remove_ideas = {
            parliamentary_republic_idea
            oligarchy_idea
            socialist_republic_idea
            constitutional_monarchy_idea
            absolute_monarchy_idea
            dictatorship_idea
            one_party_state_idea
            anarcho_communist_idea
            anarcho_capitalist_idea
            provisional_government_idea
            colonial_government_idea
        }

        add_ideas = federal_republic_idea
    }
    end_coalition = yes
}
