scripted_gui = {
    display_government_tab_gui = {
        window_name = "government_tab_window"
        context_type = player_context

        effects = {
            government_tab_close_btn_click = {
                clr_country_flag = display_government_tab
            }

            government_tab_close_btn2_click = {
                clr_country_flag = display_government_tab
            }

            government_info_btn_click = {
                set_country_flag = show_government_info
                clr_country_flag = show_government_upper_house
                clr_country_flag = show_government_lower_house
            }

            government_upper_house_btn_click = {
                set_country_flag = show_government_upper_house
                clr_country_flag = show_government_info
                clr_country_flag = show_government_lower_house
            }

            government_lower_house_btn_click = {
                set_country_flag = show_government_lower_house
                clr_country_flag = show_government_info
                clr_country_flag = show_government_upper_house
            }
		}

        triggers = {
            government_upper_house_btn_click_enabled = {
                has_elections = yes
            }

            government_lower_house_btn_click_enabled = {
                has_elections = yes
                has_country_flag = has_government_lower_house
            }
        }

        visible = {
            has_country_flag = display_government_tab
        }
    }
}
