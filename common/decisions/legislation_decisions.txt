    end_our_coalition = {

		icon = hol_exchange_intelligence_data

        visible = {
            OR = {
                has_country_flag = is_parliamentary_republic
                has_country_flag = is_constitutional_monarchy
            }
            OR = {
                has_country_flag = is_in_coalition_with_anarchists
                has_country_flag = is_in_coalition_with_totalists
                has_country_flag = is_in_coalition_with_socialists
                has_country_flag = is_in_coalition_with_social_democrats
                has_country_flag = is_in_coalition_with_liberals
                has_country_flag = is_in_coalition_with_conservatives
                has_country_flag = is_in_coalition_with_authoritarians
                has_country_flag = is_in_coalition_with_ultranationalists
            }
		}

		available = {
            has_elections = yes
            custom_trigger_tooltip = {
                tooltip = trig_in_coalition
                OR = {
                    has_country_flag = is_in_coalition_with_anarchists
                    has_country_flag = is_in_coalition_with_totalists
                    has_country_flag = is_in_coalition_with_socialists
                    has_country_flag = is_in_coalition_with_social_democrats
                    has_country_flag = is_in_coalition_with_liberals
                    has_country_flag = is_in_coalition_with_conservatives
                    has_country_flag = is_in_coalition_with_authoritarians
                    has_country_flag = is_in_coalition_with_ultranationalists
                }
            }
		}

		cost = 50

        complete_effect = {
            custom_effect_tooltip = alt_end_coalition_message
            end_coalition = yes
		}

		ai_will_do = {
			base = 1

            modifier = {
                add = 60

                check_variable = {
                    var = lower_house_leading_party_var
                    value = lower_chamber_required_seats_60_var
                    compare = greater_than_or_equals
                }

                check_variable = {
                    var = upper_house_leading_party_var
                    value = upper_chamber_required_seats_60_var
                    compare = greater_than_or_equals
                }
            }

            modifier = {
                add = 100

                OR = {
                    check_variable = {
                        var = leading_party_lower_house_total_seats
                        value = lower_chamber_required_seats_40_var
                        compare = less_than
                    }

                    check_variable = {
                        var = leading_party_upper_house_total_seats
                        value = upper_chamber_required_seats_40_var
                        compare = less_than
                    }
